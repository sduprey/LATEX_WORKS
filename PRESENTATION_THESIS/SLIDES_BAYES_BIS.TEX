\begin{slide} 
\vspace*{\fill} 
\heading{\centering{\textcolor{blue}{I. Le théorème de bayes}}}
\vspace*{\fill}
\begin{itemize}
  \item Soit A et B deux évènements, on a la relation \ : \\
  \vspace*{\fill} 
  \heading{\centering{\textcolor{red}{$\Prob{A|B}=\frac{\Prob{B|A}\Prob{A}}{\Prob{B}}$}}}
  \vspace*{\fill} 
  \item Autres présentations \ :
    \begin{itemize}
    	\item Soit A et B deux évènements, on a la relation \ : \\
  		\vspace*{\fill} 
\heading{\centering{$\Prob{A|B}=\frac{\Prob{B|A}\Prob{A}}{\Prob{B|A}\Prob{A}+\Prob{B|A^{c}}\Prob{A}^{c}}$}}
    	\vspace*{\fill} 

      \item Soit $\{A_{i}\}$ un système complet et B un évènement, on a alors la relation \ :\\
  		\vspace*{\fill} 
\heading{\centering{$\Prob{A_{i}|B}=\frac{\Prob{B|A_{i}}\Prob{A_{i}}}{\sum_{j}\Prob{B|A_{j}}\Prob{A_{j}}}$}}
  		\vspace*{\fill} 
    \end{itemize}
\end{itemize}

\vspace*{\fill}
\end{slide}

\begin{slide} 
\vspace*{\fill} 
\heading{\centering{\textcolor{blue}{II. Approche bayésienne}}}
\vspace*{\fill}
\textcolor{blue}{1. Un exemple pour commencer \ : les " faux positifs" médicaux.}

\begin{itemize}
  \item Un test permet de dépister une maladie.
  \item On sait qu'a priori la maladie touche une personne sur mille.
  \item On observe les résultats suivant sur le test \ :
  	\begin{itemize}
  		\item Si un patient a contracté la maldie, le test est positif avec une probabilité $0,99$.
  		\item Si un patient est sain, le test est négatif avec une probabilité $0,95$.
  	\end{itemize}
  \item Un patient veut connaître la probabilité a posteriori qu'il soit malade si le résultat du test est positif.
  \item On considère les évènements \ :
  	\begin{itemize}
  		\item A \ : le patient a contracté la maladie.
  		\item B \ : le test est positif.
  	\end{itemize}
  \item On applique le théorème de Bayes \ :
  	\begin{align}
  		\Prob{A|B}  & =\frac{0,99\times0,001}{0,99\times0.001+0,05\times0,999}, \nonumber \\
  								& \approx 0,019 \nonumber
  	\end{align}
  \item L'approche bayésienne nous apprend que le test n'est pas suffisamment fiable.
\end{itemize}

\vspace*{\fill}
\end{slide}

\begin{slide} 
\vspace*{\fill} 
\heading{\centering{\textcolor{blue}{II. Approche bayésienne}}}
\textcolor{blue}{2. L'analyse bayésienne.}
\vspace*{\fill}

\begin{itemize}
	\item soit les évènements \ :
		\begin{itemize}
			\item Théorie \ : la théorie H est vraie.
			\item Données \ : certaines données ont été observées.
		\end{itemize}
	\item Le théorème de Bayes s'écrit \ :

  \vspace*{\fill} 
\heading{\centering{$\Prob{\mbox{Théorie}|\mbox{Données}}=\frac{\Prob{\mbox{Théorie}}\times\Prob{\mbox{Données}|\mbox{Théorie}}}{\Prob{\mbox{Données}}}$}}
  \vspace*{\fill} 
	
	\item Estime la \textbf{probabilité a posteriori} d'une théorie au vu de résultats d'observations, si l'on peut calculer les prédictions de la théorie $\Prob{\mbox{Données}|\mbox{Théorie}}$, appelé \textbf{vraissemblance} et si l'on connaît la \textbf{probabilité a priori} de la théorie $\Prob{\mbox{Théorie}}$.
	\item Permet de comparer plusieurs théories.
	\item L'approche bayésienne compare des hypothèses aux données réelles.
	\item Elle dépend des données examinées et des données et connaissances antérieures.
\end{itemize}

\vspace*{\fill}
\end{slide}

\begin{slide} 
\vspace*{\fill} 
\heading{\centering{\textcolor{blue}{II. Approche bayésienne}}}
\textcolor{blue}{3. Les réseaux bayésiens}
\vspace*{\fill}
\begin{itemize}
  \item A chaque noeud du réseau est associée une variable aléatoire discréte et une loi de 		probabilité est associée à chaque noeud sans parent.
  \item A l'ensemble des arcs pointant sur un noeud on associe une table de probabilité 			conditionnelle.
  \item L'état d'un noeud conditionnellement à ses noeuds parents est indépendant des 				autres noeuds du réseau.
\end{itemize}
\vspace*{\fill}
\end{slide}

\begin{slide} 
\vspace*{\fill} 
\heading{\centering{\textcolor{blue}{III. Exemple d'un système à trois composants}}}
\vspace*{\fill}

Approche possible \ :
\begin{itemize}
  \item On connaît $p_{1}=\Prob{X_{1}=1},p_{2}=\Prob{X_{2}=1}$ et $p_{3}=\Prob{X_{3}=1}$ connues a priori.
  \item On considère $\Prob{\phi=1}$ connue a priori.
  \item Les vraissemblances $\Prob{\phi=1|X_{1}=1}, \Prob{\phi=1|X_{2}=1}$ et $\Prob{\phi=1|X_{3}=1}$ peuvent être obtenues par simulation, mais aussi par un calcul direct.
  \item On peut ainsi déterminer, par exemple, la probabilité a posteriori $\Prob{X_{1}=1|\phi=1}$ que le composant $1$ fonctionne quand le système fonctionne, ou encore $\Prob{X_{1}=0|\phi=0}$, la probabilité que la défaillance du composant $1$ soit l'une des causes de la défaillance du système.
\end{itemize}

\vspace*{\fill}
\end{slide}

\begin{slide} 
\vspace*{\fill} 
\heading{\centering{\textcolor{blue}{III. Exemple d'un système à trois composants}}}
\vspace*{\fill}

Autre approche possible \ :
\begin{itemize}
	\item soit les évènements \ :
		\begin{align}
		E_{1}=X_{1}\cap X_{2}\cap X_{3} \nonumber \\
		E_{2}=X_{1}\cap X_{2}\cap X^{c}_{3} \nonumber \\
		. \nonumber \\
		. \nonumber \\
		. \nonumber \\
		E_{3}=X^{c}_{1}\cap X^{c}_{3}\cap X^{c}_{3} \nonumber
		\end{align}
	\item On connaît a priori $\Prob{E_{k}=1}$ pour $k$ compris entre $1$ et $8$.
	\item Le tableau de probabilité conditionnelle donne $\Prob{\phi=1|E_{k}=1}$ pour $k$ compris entre $1$ et $8$.
	\item On calcul \ :
	$$\Prob{\phi=1}=\sum^{8}_{k=1}\Prob{\phi=1|E_{k}=1}\Prob{E_{k}=1}$$
	\item On peut ainsi déterminer les causes du fonctionnement ou de la défaillance du système.
	\item méthode trés lourde à mettre en oeuvre ($2^{n}$ possibilités pour $n$ composants).
\end{itemize}

\begin{itemize}
  \item L'approche bayésienne permet d'obtenir des informations supplémentaires.
  \item Elle aide à la prise de décision en sélectionnant la théorie la plus probable pour expliquer ce qui est observé.
