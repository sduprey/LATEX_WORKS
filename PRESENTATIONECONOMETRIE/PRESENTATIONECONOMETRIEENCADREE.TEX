\begin{slide}
\slideheading{Processus autorégressifs composés : analyse mathématique}
\textbf{\textcolor{blue}{\large Log-Laplace transformée conditionnelle : fonction affine des valeurs passées du processus}}
\newline\textbf{Définition : }\newline
Soit $(Y_t,t\geq 0)$ un processus à n dimensions et notons $\underline{Y_{t-1}}$ l'ensemble des informations jusqu'à et incluant $t-1$. Le processus Y est un processus autorégressif d'ordre p {\bf CAR(p)} si et seulement si la distribution conditionnelle de $Y_t$ sachant $\underline{Y_{t-1}}$ admet une transformée de Laplace conditionnelle du type :
\BEQ{definitionCAROrdreP}
E\left[exp(u'Y_t)|\underline{Y_{t-1}}\right]=exp\left[-a_1'(u)Y_{t-1}-...-a_p'(u)Y_{t-p}+b(u)\right]
\EEQ, où $a_p \neq 0$.
\end{slide}
\begin{slide}
\textbf{\textcolor{blue}{\large Equivalence CAR(p) et CAR(1)}}
\newline\textbf{Proposition : }\newline
Le processus $(Y_t,t\geq 0)$ est un processus $CAR(p)$ si et seulement si le processus $\widetilde{Y}_t=\left(Y'_{t},Y'_{t-1},\hdots,Y'_{t-p}\right)$ est un processus $CAR(1)$.
\BEQ{definitionNouvelleFonction}
a(v)=\left(
\begin{array}{c}
\dsp a_1(v_1)+v_2 \\
\dsp  \vdots  \\
\dsp a_{p-1}(v_1)+v_p   \\
\dsp a_{p}(v_1)\\
\end{array}
\right)
\EEQ
\end{slide}
\begin{slide}
\textbf{\textcolor{blue}{\large Processus autorégressifs composés à valeurs entières et thématique de file d'attente :}}\newline
$Y_t=Z_t+\epsilon_t$ avec $Z_t=\rho Y_{t-1}$. Problème $Z_t$ pas entier !
\BEQ{definitionAppellationCompose}
Y_t=\sum_{i=1}^{Y_{t-1}}Z_{i,t}+\epsilon_t
\EEQ, où les variables $Z_{i,t}$ suivent une loi de Bernouilli
$\mathbb{B}(1,\rho)$. Dans ce cas :
\begin{displaymath}
E[exp(-uY_t)|Y^{t-1}]=exp(-a(u)Y_{t-1}+b(u))
\end{displaymath}
\textbf{\textcolor{blue}{\large Processus autorégressifs composés à valeurs positives}}\newline
On donne une condition nécessaire et suffisante sur la transformée de Laplace d'une variable aléatoire pour qu'elle soit positive.
\BEQ{completeMonotinicite}
\forall j\in \mathbb{N}, (-1)^j\frac{\ud^j}{\ud u^j}\left[exp(b(u))\right]\geq 0
\EEQ 
\end{slide}
\begin{slide}
\textbf{\textcolor{blue}{\large Distribution invariante, prévision et stationnarité}}\newline
\newline\textbf{Proposition : }\newline
La distribution invariante $E[\exp\left(-u'Y_t\right)|\underline{Y}_{t-1}]=\exp[c(u)]$ vérifie :
\begin{displaymath}
b(u)=c(u)-c(a(u))
\end{displaymath}
Un processus stochastique CAR(1) vérifie :
\BEQ{previsionHorizonQuelconque}
E[exp(-uY_{t+h})|Y^{t}]=exp(-a^{oh}(u)Y_{t}+\sum_{k=0}^{h-1}b(a^{ok}(u)))\\
\EEQ
Soit un processus CAR(1) admettant une log-Laplace transformée invariante c.
La transformée de Laplace conditionnelle tend vers une limite indépendante de la variable conditionnante si et seulement si :
\BEQ{ergodicite}
\lim_{h\rightarrow \infty}a^{oh}(u)=0, \ \forall u.
\EEQ
\end{slide}
\begin{slide}
\textbf{\textcolor{blue}{\large Réversibilité temporelle d'un processus autorégressif composé}}\newline
\newline\textbf{Proposition : }\newline
Le processus CAR(1) est réversible si et seulement si la fonction $\psi(u,v)=c(a(u)+v)+c(u)-c(a(u))$ est une fonction symétrique de u et v. La démonstration est immédiate en exprimant la symétrie de la transformée de Laplace de la distribution jointe de ($Y_t,Y_{t-1}$) par la propriété de Markov. Quand le processus $Y_t$ est réversible :
\begin{displaymath}
i) a(u) =\left( \frac{\ud c}{\ud u}\right)^{-1}\left[\frac{\ud a}{\ud u}(0)\left(\frac{\ud c}{\ud u}(u)-\frac{\ud c}{\ud u}(0) \right)+\frac{\ud c}{\ud u}(0)\right]
\end{displaymath}
ii) La fonction $\gamma(u)=\frac{\ud^2 c}{\ud u^2}o(\frac{\ud c}{\ud u})^{-1}$ est quadratique.
\end{slide}
\begin{slide}
\textbf{\textcolor{blue}{\large Opérateur espérance conditionnelle}}\newline
\newline\textbf{Proposition : }\newline
\BEQ{momentsConditionnels}
E[Y_t^n|Y_{t-1}]=P_n(Y_{t-1})
\EEQ
, où $P_n$ est un polynôme de degré n dont le coefficient de plus haut degré est $[\frac{\ud a}{\ud u}(0)]^n$. Considérons l'opérateur d'espérance conditionnelle $\psi \rightarrow T\psi $ défini par :
\BEQ{definitionOperateurConditionnel}
T\psi(y)=E[\psi(Y_t)|Y_{t-1}=y]
\EEQ
Cet opérateur admet les valeurs propres réels $\lambda_n=[\frac{\ud a}{\ud u}(0)]^n$, $n\geq 0$ et pour fonctions propres associées des polynômes $P_n$ de degré $n$.
\end{slide}
\begin{slide}
\textbf{\textcolor{blue}{\large Expression des densités conditionnelles dans le cas de la réversibilité}}\newline
\newline\textbf{Proposition : }\newline
Supposons $\vert \frac{\ud a}{\ud u}(0) \vert < 1$. Pour un processus CAR(1) stationnaire et réversible, les fonctions propres $P_n$, $n\geq 0$, de l'opérateur espérance conditionnelle sont orthogonales pour le produit scalaire associé à la distribution invariante $f$. De plus :
\BEQ{densiteReversibleStationnaire}
f\left(y_t\vert y_{t-1}\right)=f(y_t)\left[1+\sum_{n=1}^{\infty}\left[\frac{\ud a}{\ud u}(0)\right]^n P_n(y_t)P_n(y_{t-1})\right]
\EEQ, où $P_n$ est la base orthogonale des fonctions polynômiales propre de l'opérateur conditionnel d'espérance.
\BEQ{densiteReversibleStationnaireHorizonH}
f_h\left(y_t\vert y_{t-h}\right)=f(y_t)\left[1+\sum_{n=1}^{\infty}\left[\frac{\ud a}{\ud u}(0)\right]^{hn} P_n(y_t)P_n(y_{t-h})\right]
\EEQ
\end{slide}
\begin{slide}
\textbf{\textcolor{blue}{\large Représentation espace-états, filtrage et lissage}}\newline
\BEQ{nonlinearrepresentation}
Y_t = Z_t + \epsilon_t 
\EEQ
\BEQ{nonlinearrepresentationdeux}
Z_t=\alpha (Y_{t-1},\eta_t)
\EEQ
\newline\textbf{Proposition : }\newline
{\small
\begin{enumerate}
\item Les variables $Z_t$, $t$ variant, sont indépendants conditionnellement au processus observable.
\item La distribution conditionnelle de $Z_t$ sachant toutes les valeurs de $Y_t$ coïncide avec la distribution conditionnelle de $Z_t$ sachant $Y_{t-1}$ et $Y_t$ seulement (réversibilité et propriété de Markov). Cette distribution filtrante est donnée par :
\BEQ{filteringdistribution}
l(z_t \vert y_{t-1},y_t)= \frac{g(z_t\vert y_{t-1}) h(y_t-z_t)}{\int g(z\vert y_{t-1}) h(y_t-z)\ud z}
\EEQ
\item La distribution lissante de $\epsilon_t$ suit directement puisque $\epsilon_t=y_t-z_t$.
\end{enumerate}
}
\end{slide}
\begin{slide}
\textbf{\textcolor{blue}{\large Les différents processus CAR réversibles classés selon l'équation caractéristique $\beta_0 +\beta_1 x+\beta_2 x^2$ associé à l'équation de Ricatti}}
\small{
\newline $\bullet$  $\beta_1=\beta_2=0$ : les processus de Poisson composés
\newline $\bullet$  $\beta_2=0$ : les processus gaussiens autorégressifs.
\newline $\bullet$  $\beta_2\neq 0$ et $\beta_1^2-4\beta_0\beta_2 = 0$ : les processus $\gamma$-composés.
\newline $\bullet$  $\beta_2\neq 0$ et $\beta_1^2-4\beta_0\beta_2 > 0$ : les processus de Bernouilli à régime changeant.
\newline $\bullet$  $\beta_2\neq 0$ et $\beta_1^2-4\beta_0\beta_2 < 0$  : processus à $\gamma$ fonction quadratique et racines complexes conjuguées.
}
\end{slide}
\begin{slide}
\textbf{\textcolor{blue}{Inférence statistique : modèle contraint}}
\BEQ{unconstrainedModelisaiont}
E[\exp(-uY_t)|Y_{t-1}]=L^o(u,y)=\exp (-a(u)y+b(u))
\EEQ
\newline\textbf{Proposition : }\newline
\BEQ{constrainedEstimator}
\left[\hat{a}_T(u),\hat{b}_T(u)\right]=\ arg \ \min_{a,b}\sum_{t=1}^{T}\left[\exp(-u y_t)-\exp(ay_{t-1}+b)\right]^2,
\EEQ
Sous les hypothèses de régularité standards et sous la condition que le processus CAR(1) soit bien défini, l'estimateur $\left[\hat{a}_T(u),\hat{b}_T(u)\right]'$ est asymptotiquement normal et tel que :
\BEQ{normalConvergence}
\sqrt{T}\left(\left(
\begin{array}{c}
\dsp \hat{a}_T(u) \\
\dsp  \hat{b}_T(u)\\
\end{array}
\right)-\left(
\begin{array}{c}
\dsp a(u) \\
\dsp  b(u)\\
\end{array}
\right)\right)\rightarrow N\left(0,\Omega(u)\right),
\EEQ
où $\Omega(u)=J^{-1}(u)I(u)J^{-1}(u)$
\end{slide}
\begin{slide}
\textbf{\textcolor{blue}{Inférence statistique : modèle non contraint}}
\newline\textbf{Proposition : }\newline
{\small
La transformée de Laplace non contrainte peut être estimée par l'estimateur de Nadaraya-Watson :
\BEQ{}
\hat{L}_T(u,y)=\frac{\sum_{t=2}^{T}\exp (-uy_t)K_{h_T}(y_{t-1}-y)}{\sum_{t=2}^{T}K_{h_T}(y_{t-1}-y)}, \ \ u\in I, y\ variant,
\EEQ
L'estimateur est consistent et asymptotiquement normal 
\BEQ{nadarayaconvergence}
\sqrt{Th_T}\left[\hat{L}_T(u,y)-L(u,y)\right]\rightarrow N(0,\Sigma(u,y)),
\EEQ
où :
\BEQ{defSigma}
\Sigma(u,y)=\frac{1}{f(y)}\int K^2(v) dv V\left[\exp (-u Y_t)|Y_{t-1}=y\right]
\EEQ
\begin{displaymath}
\frac{L(2u,y)-[L(u,y)]^2}{f(y)}\int K^2(v) dv 
\end{displaymath}
La qualité de l'estimation peut alors être évaluée en considérant le résidu fonctionnel de $\hat{L}_T(u,y)-\hat{L}^0_T(u,y)$, u et y variant.
}
\end{slide}
\begin{slide}
\slideheading{Application "FTD Basket", "Cycle effect" et "Loss given Default"}
\textbf{\textcolor{blue}{\large Les processus $\gamma$-composés}}\newline
On peut démontrer que la log-transformée de Laplace de ce processus s'écrit :
\BEQ{logtransformeeGammaa}
E_{t-1}\left[\exp (u Y_t)\right]=(1-uc_t)^{-\delta}\exp\left(\frac{c_t u}{1-c_t u}\beta_t Y_{t-1}\right).
\EEQ
La log-transformée de Laplace vérifie donc l'hypothèse affine avec pour fonction $a(u)=\frac{c_t u}{1-c_t u}\beta_t$ et $b(u)=-\delta\log (1-c_t u)$.\newline
Dans le cadre où $c$, $\delta$ et $\gamma$ sont des constantes, la prévision de Y à l'ordre h est donné par la formule analytique suivante :
\BEQ{ORDREHPREVISIONGAMMA}
E_{t}\left[\exp (u Y_{t+h})\right]=\exp\left(Y_t\frac{\rho^h u}{1-uc\frac{1-\rho^h}{1-\rho}}-\delta\log\left(1-uc\frac{1-\rho^h}{1-\rho}\right)\right)
\EEQ
\end{slide}

\begin{slide}
\textbf{\textcolor{blue}{\large Modélisation du taux de recouvrement}}\newline
{\tiny
\begin{displaymath}
C*_i(t,t+h)=\sum_{k=1}^{h}E_t\left\{\Pi_{j=0}^{k-2}M_{t+j,t+j+1}\pi_{t+j,t+j+1}M_{t+k-1,t+k}\left(1-\pi_{t+k-1,t+k}\right)R_{t+k-1,t+k}\right\}
\end{displaymath}
\begin{displaymath}
+ E_{t}\left\{\Pi_{j=0}^{h-1}(M_{t+j,t+j+1}\pi_{t+j,t+j+1})\right\},
\end{displaymath}
}
{\small
\textbf{Hyopthèse : }
Les taux de recouvrement pour les différentes entreprises sont indépendants conditionnellement aux facteurs $Z$ et $Z_i$. Le taux de recouvrement pour l'entreprise i est tel que :
\BEQ{tauxDeRecouvrement}
R_{t+j,t+j+1}= \exp(-(\delta_{t+1} + \epsilon'_{t+1} Z_{t+1}+\theta'_{t+1}Zi_{t+1}))=\exp(-\omega'_t),
\EEQ
où $\omega'_t= \delta_t + \epsilon'_t Z_t + \theta'_t Zi_t$ est l'intensité de recouvrement et  $\delta_t$,  $\epsilon'_t$ , $\theta'_t$ des fonctions de l'information contenue dans les facteurs  $Z_t$ et $Z_i$. Ce sont aussi des paramètres de sensibilités pour le recouvrement.
Les taux de recouvrement étant compris entre 0 et 1, les sensibilités de recouvrement doivent vérifier comme les intensités de survie : $\omega'_t\ge 0$. 
}
